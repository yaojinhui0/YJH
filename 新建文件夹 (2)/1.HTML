DATA SEGMENT

VARX DW 10D

VARY DW 5D

VARZ DW 0

DATA ENDS

CODE SEGMENT

ASSUME CS:CODE,DS:DATA

START:

MOV AX,DATA ; 将数据段的地址加载到 AX

MOV DS,AX ; 将数据段地址传送到 DS

MOV AX,0 ; 清空 AX 寄存器

MOV AX,VARX ; 将 VARX 的值加载到 AX 寄存器

SHL AX,1 ; 左移 1 位，相当于乘以 2

ADD AX,VARX ; AX = AX + VARX，等价于 AX = 2 * VARX + VARX

ADD AX,VARY ; AX = AX + VARY，结果是 AX = 3 * VARX + VARY

SUB AX,5 ; AX = AX - 5

SHR AX,1 ; 右移 1 位，相当于除以 2
